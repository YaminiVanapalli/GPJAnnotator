
SELECT DISTINCT K.KITREF, K.KITID, K.KITFNAME, 
EU.ELTID, EU.ELTFNAME, EU.ELTDESC, 
GM.GTYPE, GM.GSTATUS 
FROM IGL_KIT_USE KU, IGL_ELEMENT_USE EU, IGL_KIT K, GFX_ELEMENT GE, GFX_MASTER GM 
WHERE KU.SHYR = 2001 
AND KU.AN8_SHCD = 13973 
AND KU.AN8_CUNO = 1159 
AND KU.SHYR = EU.SHYR 
AND KU.AN8_SHCD = EU.AN8_SHCD 
AND KU.AN8_CUNO = EU.AN8_CUNO 
AND KU.KITUSEID = EU.KITUSEID 
AND EU.SHSTATUS <> 3 
AND EU.KITID = K.KITID 
AND EU.ELTID = GE.ELTID 
AND GE.GID = GM.GID 
AND GM.GSTATUS IN (10, 20, 30) 
ORDER BY K.KITREF, EU.ELTFNAME, GM.GTYPE, GM.GSTATUS;
